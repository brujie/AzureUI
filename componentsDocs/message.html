    # Message

    消息提示

    # 示例


    # 代码

        import { Message } from 'azure-ui'
    Vue.prototype.$Message = Message;

    methods:{
        click(){
            this.$Message.success('成功提示!');
            this.$Message.error({
                message:'错误信息!',
                duration:2000,
                center:true,
                verticalOffset:50
            })
        }
    }

    # 属性

    参数 说明 类型 是否必填 默认值
    message 文字信息 Number false 提示信息!
    duration 持续时间 Number false 1500
    center 是否居中 Boolean false false
    verticalOffset 距离顶部高度 Number false 20
